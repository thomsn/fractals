from stl import mesh
import math
import numpy


vectors = numpy.array([
[
    [1, 0, 0],
    [0, 1, 0],
    [0, 0, 1]
],                         
                           [
    [-1, 0, 0],
    [0, 1, 0],
    [0, 0, 1]
],                               
                 [
    [-1, 0, 0],
    [0, -1, 0],
    [0, 0, 1]
],          
                           [
    [1, 0, 0],
    [0, -1, 0],
    [0, 0, 1]
],                                       
])

spike_x = 3
spike_y = 3
distribution = 2
num_spikes = spike_x * spike_y
spikes = vectors.copy()
for x in range(spike_x):
    for y in range(spike_y):
        if x == 0 and y == 0:
            continue
        spike = vectors.copy()
        for vertex in spike:
            for point in vertex:
                point[0] += x * distribution
                point[1] += y * distribution
        spikes = numpy.concatenate([spikes, spike])


data = numpy.zeros(len(spikes), dtype=mesh.Mesh.dtype)
mesh = mesh.Mesh(data)

# x y z
data['vectors'] = spikes

from mpl_toolkits import mplot3d

# Create a new plot
figure = pyplot.figure()

# Auto scale to the mesh size
scale = numpy.concatenate(mesh).flatten(-1)
axes.auto_scale_xyz(scale, scale, scale)

# Show the plot to the screen
pyplot.show()

mesh.save('spiky_surface.stl')
